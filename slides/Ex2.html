Group by

GROUP BY: aggregate many rows into 1

How many products we sold?


            SELECT
                productId,
                sum(Quantity) as total_sales
            FROM OrderDetails
            GROUP BY productID;
Average product per order?

            SELECT
                productId,
                sum(Quantity) as total_sales,
                avg(Quantity) as avg_sales,
            FROM OrderDetails
            GROUP BY productID;
